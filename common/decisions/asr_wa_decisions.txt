decision_asr_wa_wilderness_expand = {
  owned_planets_only = yes
  enactment_time = 360

  icon = wilderness_planet_expand

  potential = {
    owner? = { is_wilderness_empire = yes }
    is_artificial = no
    OR = {
      planet_size < 21
      AND = {
        owner = { has_tradition = tr_purity_stage_3_swap }
        planet_size < 26
      }
    }
  }

  resources = {
    category = decisions
    cost = {
      # 25%的额外生物质花费以换取4倍的扩大速度
      biomass = 25
      mult = value:asr_wa_expand_planet_mod
    }
    cost = {
      trigger = {
        owner = { is_ai = no }
      }
      influence = 200
    }
  }
  inline_script = { script = asr_wa/shared/biomass_spent_queued_unqueued }

  effect = {
    change_planet_size = 4
  }

  ai_weight = {
    weight = 0
    modifier = {
      weight = 1000
      free_district_slots <= 0
      owner = {
        resource_stockpile_compare = {
          resource = biomass
          value > 4000
        }
      }
    }
  }
}

decision_asr_wa_wilderness_world = {
  owned_planets_only = yes
  # icon = decision_wilderness_build_colony_ship
  enactment_time = 7200

  potential = {
    owner? = {
      is_wilderness_empire = yes
    }
    asr_wa_planet_is_wilderness_arcology = yes
  }
  allow = {
    has_fully_upgraded_capital = yes
    OR = {
      has_modifier = mastery_of_nature
      has_modifier = cultivated_worldscaping_modifier
    }
    planet_size >= 25
    free_district_slots <= 0
    num_uncleared_blockers <= 0
    owner? = {
      OR = {
        has_ascension_perk = ap_mastery_of_nature
        is_environmental_architects_empire = yes
      }
      has_ascension_perk = ap_world_shaper
      has_technology = tech_mega_engineering
    }
  }

  resources = {
    category = decisions
    cost = {
      food = 4000
      biomass = 400
      mult = trigger:planet_size
    }
    cost = {
      trigger = {
        owner = { is_ai = no }
      }
      influence = 200
    }
  }
  inline_script = { script = asr_wa/shared/biomass_spent_queued_unqueued }

  effect = {
    hidden_effect = {
      set_carrier_flag = asr_wa_pf_wilderness_world
      every_deposit = {
        limit = {
          NOR = {
            has_deposit_category = deposit_cat_rare
            has_deposit_category = deposit_cat_rare_resource
            has_deposit_category = deposit_cat_rare_blocker
          }
        }
        remove_deposit = yes
      }
    }
    add_modifier = {
      modifier = asr_wa_wilderness_world
      days = -1
    }
    custom_tooltip = decision_asr_wa_wilderness_world_tt
    create_message = {
      type = TERRAFORM_MESSAGE_TYPE
      localization = decision_asr_wa_wilderness_world_message
      target = this
      days = 30
      variable = {
        scope = this
        type = name
        localization = PLANET
      }
    }
  }

  ai_weight = {
    weight = 0
    modifier = {
      weight = 1000
      owner = {
        resource_stockpile_compare = {
          resource = biomass
          value > 15000
        }
        OR = {
          resource_income_compare = {
            resource = food
            value > 1000
          }
          resource_stockpile_compare = {
            resource = food
            value > 128000
          }
        }
      }
    }
  }
}
