# 掌控自然飞升额外加成
asr_wa_mastery_of_nature_effect = {
  potential = {
    is_wilderness_empire = yes
  }

  option = {
    name = asr_wa_p_mastery_of_nature_none
    potential = {
      NOR = {
        has_ascension_perk = ap_mastery_of_nature
        is_environmental_architects_empire = yes
      }
    }
    valid = {
      NOR = {
        has_ascension_perk = ap_mastery_of_nature
        is_environmental_architects_empire = yes
      }
    }

    ai_weight = { weight = 0 }
  }

  option = {
    name = asr_wa_p_mastery_of_nature_valid_1
    potential = { always = yes }
    valid = {
      OR = {
        has_ascension_perk = ap_mastery_of_nature
        is_environmental_architects_empire = yes
      }
      asr_wa_policy_building_capacity_and_time_check = {
        POLICY_FLAG = asr_wa_p_mastery_of_nature_valid_1
        CAP = -4
        TIME = -0.5
        POLICY_FLAG_ALT = asr_wa_p_mastery_of_nature_valid_2

        TIME_ALT = -0.25
      }
    }

    policy_flags = { asr_wa_p_mastery_of_nature_valid_1 }

    modifier = {
      planet_structures_cost_mult = -0.10
      planet_building_capacity_add = -4

  option = {

        TIME_ALT = 0

    policy_flags = { asr_wa_p_mastery_of_nature_valid_2 }

    modifier = {
      planet_structures_cost_mult = -0.10
      planet_building_capacity_add = -2
      building_time_mult = -0.25
    }

    ai_weight = { weight = 10 }
  }

# 纯净传统额外加成
asr_wa_tr_purity_effect = {
  potential = {
    is_wilderness_empire = yes
    OR = {
      has_tradition = tr_purity_adopt # 被释放的附庸会复制你的传统
      has_ascension_perk = ap_engineered_evolution
    }
  }

  option = {
    name = asr_wa_p_genetic_ascension

    potential = {
      NOR = {
        has_tradition = tr_purity_finish
        has_tradition = tr_purity_adopt
      }
      has_ascension_perk = ap_engineered_evolution
    }

    valid = {
      NOR = {
        has_tradition = tr_purity_finish
        has_tradition = tr_purity_adopt
      }
      has_ascension_perk = ap_engineered_evolution
    }

    modifier = {}

    AI_weight = {
      weight = 10
    }
  }

  option = {
    name = asr_wa_p_purity_adopt

    potential = {
      NOT = { has_tradition = tr_purity_finish }
      has_tradition = tr_purity_adopt
    }

    valid = {
      NOT = { has_tradition = tr_purity_finish }
      has_tradition = tr_purity_adopt
    }

    modifier = {
      leader_initial_skill = 2
    }

    AI_weight = {
      weight = 10
    }
  }

  option = {
    name = asr_wa_p_purity_finish

    potential = {
      has_tradition = tr_purity_finish
    }

    valid = {
      has_tradition = tr_purity_finish
    }

    modifier = {
      leader_initial_skill = 2
      councilor_skill_add = 2
    }

    AI_weight = {
      weight = 10
    }
  }
}
