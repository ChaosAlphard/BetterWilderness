  icon = "avatar_chamber_1"
  ruined_icon = "wilderness_building_ruined"
  base_buildtime = @buildtime_1
  category = amenity
  can_build = yes
  can_demolish = yes

  building_sets = {
    government
  }

  inline_script = { script = buildings/wilderness_buildings_potential_destroy }
  allow = {
    hidden_trigger = { exists = owner }
    OR = {
      owner = { is_ai = no }
      free_amenities < 1000
      NOR = {
        has_building = building_avatar_chamber_1
        has_building = building_avatar_chamber_2
        has_building = building_avatar_chamber_3
        has_building = building_avatar_chamber_4
      }
    }
  }

  convert_to = {
    building_drone_storage
    building_hive_warren
    building_communal_housing
    building_luxury_residence
  }

  prerequisites = {
    tech_wilderness_node
  }

  resources = {
    category = planet_buildings
    inline_script = {
      script = asr_wa/building/shared_cost_food
      BIOMASS = @wa_b1_biomass
      FOOD = @wa_b1_foods
      MINERAL = @wa_b1_minerals
    }
    upkeep = {
      food = @b1_upkeep
    }
  }

  upgrades = {
    building_avatar_chamber_2
  }

  planet_modifier = {
    planet_housing_add = @wa_t1_housing
    planet_amenities_no_happiness_add = @wa_t1_amenities
  }

  inline_script = {
    script = buildings/on_all_wilderness_buildings_districts
  }

  ai_estimate_without_unemployment = yes
}
