# 荒野没有维护子个体, 人口太多AI处理不了会导致舒适度爆炸
# 所以区划对AI提供更少住房换取舒适度
triggered_planet_modifier = {
  potential = { always = yes }
  planet_housing_add = 400
}
triggered_planet_modifier = {
  potential = {
    exists = owner
    owner = {
      has_tradition = tr_prosperity_public_works
    }
  }
  planet_housing_add = 200
}
triggered_planet_modifier = {
  potential = {
    exists = owner
    owner = { is_ai = yes }
  }
  planet_housing_add = -200
  planet_amenities_no_happiness_add = 200
}
