# mult 在 triggered_planet_pop_group_modifier_for_all 中不起效
# 可以考虑每个阶段(一共10个)的加成都写出来, 或者提供新的岗位, 岗位提供人口组装
# 模拟组装岗位
triggered_planet_modifier = {
  potential = {
    exists = owner
    owner = { is_wilderness_empire = yes }
    count_owned_pop_amount = {
      limit = { asr_wa_is_normal_wilderness_pop = yes }
      count <= @wa_bonus_growth_max_pops
    }
  }
  job_asr_wa_simulate_spawning_drone_add = 100
  mult = value:asr_wa_simulate_spawning_drone_from_pops
}
