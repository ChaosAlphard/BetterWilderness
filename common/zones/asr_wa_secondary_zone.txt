# 工业集群
zone_asr_wa_factory = {
  icon = GFX_district_specialization_factory
  base_buildtime = @zone_buildtime
  potential = {
    asr_wa_planet_is_wilderness_arcology = yes
    exists = owner
    owner = {
      is_wilderness_empire = yes
      OR = {
        has_technology = tech_luxuries_1
        country_uses_consumer_goods = yes
      }
    }
  }
  unlock = {
    hidden_trigger = {
      asr_wa_planet_is_wilderness_arcology = yes
      exists = owner
      owner = {
        is_wilderness_empire = yes
        OR = {
          has_technology = tech_luxuries_1
          country_uses_consumer_goods = yes
        }
      }
    }
  }
  resources = {
    category = planet_zones
    cost = {
      minerals = @wa_zone_minerals
      food = @wa_zone_minerals
      trigger = { asr_wa_planet_use_food_build_building = no }
    }
    cost = {
      food = @wa_zone_foods
      trigger = { asr_wa_planet_use_food_build_building = yes }
    }
    upkeep = { consumer_goods = 0.01 }
  }

  zone_sets = { gaia_rural }

  included_building_sets = { factory }

  swap_type = district_hive_factory
  swap_type_weight = 10

  max_buildings = 3
  planet_modifier = {
    zone_building_slots_add = 3
  }

  inline_script = {
    script = asr_wa/zone/job/artisan_add
    BASE = @doubled_scaling_district_1_job
  }
}

  base_buildtime = @zone_buildtime
  icon = GFX_district_specialization_exotic_gases
  planet_limit = 1

  # 由于这个区域只出现在 gaia_rural 的 zone_sets 中
  # 所以理论上可以不用判断 owner 是否是荒野
  potential = {
    asr_wa_planet_is_wilderness_arcology = yes
    has_exotic_gases_deposit = yes
  }
  unlock = {
    has_exotic_gases_deposit = yes
    hidden_trigger = { exists = owner }
    owner = { has_technology = tech_mine_exotic_gases }
  }

  inline_script = { script = asr_wa/zone/cost }

  max_buildings = 3

  planet_modifier = {
    zone_building_slots_add = 3
    planet_farmers_exotic_gases_produces_add = 0.5
    planet_farmers_food_produces_add = -3
  }
  inline_script = {
    script = asr_wa/zone/job/farmer_add
    BASE = @doubled_scaling_district_1_job
  }

  swap_type = district_asr_wa_swap_exotic_gases
  swap_type_weight = 5

  zone_sets = {
    gaia_rural
  }

  included_building_sets = {
    farming
    fallen_empire_farming
    hydroponics
  }

  ai_priority = 10
}
