asr_wa_planet_use_mineral_build_building = {
  OR = {
    owner? = {
      has_policy_flag = asr_wa_p_building_cost_default
    }
    # owner 不存在时, 或者其他特殊情况下的保险
    AND = {
      asr_wa_planet_use_mixed_build_building = no
      asr_wa_planet_use_food_build_building = no
    }
  }
}
asr_wa_planet_use_mixed_build_building = {
  owner? = {
    has_policy_flag = asr_wa_p_building_cost_mixed
    is_ai = no  # 当前版本(4.3)AI不会用食物建造特定建筑以外的建筑和区划
  }
}
asr_wa_planet_use_food_build_building = {
  owner? = {
    has_policy_flag = asr_wa_p_building_cost_food
    is_ai = no
  }
}

# @[...]用在 scripted_trigger 和 scripted_effect 中时, 需要替换为@\[...]的形式
asr_wa_policy_building_capacity_and_time_check = {
  OR = {
    AND = {
      check_modifier_value = {
        modifier = planet_building_capacity_add
        value >= @\[$CAP$ * -1]  # 0+($CAP$ * -1)
      }
      check_modifier_value = {
        modifier = building_time_mult
        value > @\[-1 + ($TIME$ * -1)]
      }
    }
    AND = {
      has_policy_flag = $POLICY_FLAG$
      check_modifier_value = {
        modifier = planet_building_capacity_add
        value >= 0
      }
      check_modifier_value = {
        modifier = building_time_mult
        value > -1
      }
    }
    AND = {
      has_policy_flag = $POLICY_FLAG_ALT$
      check_modifier_value = {
        modifier = planet_building_capacity_add
        value >= @\[$CAP_ALT$ * -1]
      }
      check_modifier_value = {
        modifier = building_time_mult
        value > @\[-1 + ($TIME_ALT$ * -1)]
      }
    }
  }
}
