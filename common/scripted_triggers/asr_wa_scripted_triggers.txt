asr_wa_is_normal_wilderness_pop = {
  has_trait = trait_wilderness
  has_population_control = {
    type = population_control_no
    country = owner
  }
}

asr_wa_planet_use_mineral_build_building = {
  OR = {
    AND = {
      exists = owner
      owner = {
        has_policy_flag = asr_wa_pf_building_cost_default
      }
    }
    # owner 不存在时, 或者其他特殊情况下的保险
    AND = {
      asr_wa_planet_use_mixed_build_building = no
      asr_wa_planet_use_food_build_building = no
    }
  }
}
asr_wa_planet_use_mixed_build_building = {
  exists = owner
  owner = {
    has_policy_flag = asr_wa_pf_building_cost_mixed
    is_ai = no  # 当前版本(4.3.3)AI不会用食物建造特定建筑以外的建筑和区划
  }
}
asr_wa_planet_use_food_build_building = {
  exists = owner
  owner = {
    has_policy_flag = asr_wa_pf_building_cost_food
    is_ai = no
  }
}

# 兼容其他Mod, 重写这个方法以让其他星球类型也被视为盖亚星球
asr_wa_is_wilderness_advanced_planet = {
  is_planet_class = pc_gaia
}

asr_wa_planet_is_wilderness_arcology = {
  hidden_trigger = { exists = planet }
  planet = {
    asr_wa_is_wilderness_advanced_planet = yes
    [[NOT_WILDERNESS_WORLD] NOT = {
      has_planet_flag = asr_wa_pf_wilderness_world
    } ]
  }
}

asr_wa_planet_is_wilderness_world = {
  hidden_trigger = { exists = planet }
  planet = {
    asr_wa_is_wilderness_advanced_planet = yes
    has_planet_flag = asr_wa_pf_wilderness_world
  }
}

# 兼容其他Mod, 重写这个方法以在其他星球上提供等效盖亚星球的岗位
# 判断当前行星是否应该具有和 荒野盖亚星球 等效的岗位数量
asr_wa_planet_is_wilderness_arcology_equivalent = {
  asr_wa_is_wilderness_advanced_planet = yes
  NOT = { has_planet_flag = asr_wa_pf_wilderness_world }
}
# 判断当前行星是否应该具有和 荒野重塑自然 等效的岗位数量
asr_wa_planet_is_wilderness_world_equivalent = {
  asr_wa_is_wilderness_advanced_planet = yes
  has_planet_flag = asr_wa_pf_wilderness_world
}
asr_wa_planet_is_wilderness_arcology_or_world_equivalent = {
  OR = {
    asr_wa_planet_is_wilderness_arcology_equivalent = yes
    asr_wa_planet_is_wilderness_world_equivalent = yes
  }
}

asr_wa_has_any_workforce = {
  count_owned_workforce = {
    limit = { has_job_type = $JOB$ }
    count > 0
  }
}
